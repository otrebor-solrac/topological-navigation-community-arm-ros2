% Lámina 4: Planteamiento del problema
{
\usebackgroundtemplate{\includegraphics[width=\paperwidth,height=\paperheight]{figuras/pptx_images/Information_1.jpg}}
\begin{frame}[t]
    \vspace{1.2cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Planteamiento del problema}} \par
    \vspace{0.4cm}
    \begin{block}{La abstracción del movimiento}
        El manipulador antropomórfico no se mueve en un plano infinito, sino en un \textbf{espacio de configuración con estructura toroidal ($T^n$)} \parencite{Munkres2000}. 
    \end{block}
    \vspace{0.3cm}
    \textbf{Retos principales:}
    \begin{enumerate}
        \item Transformar esta geometría abstracta en un entorno de cálculo numérico \parencite{farber2008}.
        \item Contrastar dos filosofías de exploración geométrica: 
        \begin{itemize}
            \item \textbf{A*:} Búsqueda del camino óptimo (más corto) en cuadrícula \parencite{Hart1968}.
            \item \textbf{RRT:} Exploración rápida mediante muestreo de puntos aleatorios \parencite{LaValle1998}.
        \end{itemize}
        \item Carencia de un entorno de \textit{software caja blanca} para experimentar la viabilidad topológica de las trayectorias.
    \end{enumerate}
\end{frame}
}

% Lámina 5: Pregunta de Investigación e Hipótesis
{
\usebackgroundtemplate{\includegraphics[width=\paperwidth,height=\paperheight]{figuras/pptx_images/Information_1.jpg}}
\begin{frame}[t]
    \vspace{1.2cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Pregunta de investigación e hipótesis}} \par
    \vspace{0.4cm}
    \textbf{Pregunta de investigación:} \\
    \textit{¿De qué manera es posible modelar el espacio de configuración mediante una variedad toroidal, integrando recubrimientos por bolas abiertas, para permitir el estudio comparativo de eficiencia entre los algoritmos A* y RRT?}
    
    \vspace{0.4cm}
    \textbf{Hipótesis:} \\
    La formalización toroidal y el modelado por bolas abiertas permitirán resolver la conectividad en el espacio libre. Se anticipa que:
    \begin{itemize}
        \item \textbf{A*:} Garantizará la trayectoria óptima respecto a la métrica seleccionada, siendo ideal para análisis de precisión.
        \item \textbf{RRT:} Ofrecerá completitud probabilística y mayor eficiencia computacional en entornos de alta complejidad geométrica.
    \end{itemize}
\end{frame}
}

    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Estado del arte: Paradigmas y brecha tecnológica}} \par
    \vspace{0.4cm}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Evolución del paradigma}
            \begin{itemize}
                \item \small \textbf{Tradicional:} Nubes de puntos y mallas discretas (\textit{ceguera estructural}).
                \item \small \textbf{Moderna:} Variedades topológicas ($T^n$) y geometrías convexas (\textit{Sphere World}) \parencite{radhakrishnan2024state}.
                \item \small \textbf{Algoritmos:} Trade-off entre optimalidad ($A^*$) y velocidad ($RRT$) \parencite{belmont2026many}.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.5\textwidth}
            \textbf{La brecha de investigación}
            \begin{itemize}
                \item \small \alert{Vacío pedagógico:} Uso excesivo de \enquote{cajas negras} (MoveIt2) en hardware de bajo costo \parencite{semwal2024manipulator}.
                \item \small \alert{Crisis de reproducibilidad:} Dependencias de software que expiran rápidamente.
                \item \small \alert{Propuesta:} Un entorno de \enquote{caja blanca} con Docker y vinculación física real \parencite{cervera2023run}.
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{0.4cm}
    \begin{block}{Oportunidad de la tesis}
        Integrar el rigor matemático de las variedades toroidales en una plataforma educativa transparente y reproducible.
    \end{block}
\end{frame}

\section{Avances teóricos}
{
\usebackgroundtemplate{\includegraphics[width=\paperwidth,height=\paperheight]{figuras/pptx_images/Information_1.jpg}}
\begin{frame}[t]
    \vspace{1.1cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Robótica topológica y el toro $T^n$}} \par
    \vspace{0.2cm}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \begin{itemize}
            \item \textbf{Construcción:} El estado articular reside en el producto cartesiano de circunferencias ($S^1$).
            \item \textbf{Compacidad:} Por el Teorema de Tíconov, la variedad garantiza que los algoritmos no diverjan.
            \item \textbf{Topología cociente:} Identifica los bordes del espacio. El ángulo $359^\circ$ es continuo con el $1^\circ$.
        \end{itemize}
        
        \vspace{0.1cm}
        \begin{block}{Definición Topológica}
            \small \centering $\mathbb{T}^n = \underbrace{S^1 \times S^1 \times \dots \times S^1}_{n} \cong \mathbb{R}^n / \mathbb{Z}^n$
        \end{block}
        
        \vspace{0.1cm}
        {\scriptsize \color{PPTGrey} \textit{* Una \textbf{Variedad} es un espacio que localmente parece plano, pero cuya estructura global se cierra sobre sí misma (como una dona o la Tierra).}}

        
        \column{0.45\textwidth}
        \begin{center}
            \includegraphics[width=0.65\textwidth]{figuras/c_space_3d.pdf}
            
            \vspace{0.05cm}
            \textit{\tiny \textbf{Fig 1.} Modelado 3D del espacio de configuración.}
        \end{center}
    \end{columns}
\end{frame}
}

    \vspace{1.1cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Geometría y recubrimientos}} \par
    \vspace{0.2cm}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \begin{itemize}
            \item Transición de obstáculos en $\mathbb{R}^3$ a \enquote{agujeros} en el espacio de configuración ($\mathcal{C}_{obs}$ ) .
            \item \textbf{Aproximación esférica:} Se reemplazan las cajas AABB de la ingeniería clásica por \textbf{uniones finitas de bolas abiertas}.
            \item \textbf{Ventaja:} Optimiza el costo de consultas de distancia $L_2$ y garantiza consistencia topológica estricta.
        \end{itemize}

        \vspace{0.1cm}
        \begin{block}{Recubrimiento Esférico}
            \small \centering $\mathcal{C}_{obs} \approx \bigcup_{i=1}^{k} \mathcal{B}(c_i, r_i), \quad \mathcal{B} = \{ q \in \mathcal{C} \mid \|q - c_i\| < r_i \}$
        \end{block}
        
        \column{0.45\textwidth}
        \begin{center}
            \resizebox{0.70\textwidth}{!}{\input{figuras/bolas_vs_aabb.tex}}
            
            \vspace{0.05cm}
            \textit{\tiny \textbf{Fig 2.} Comparativa de Volúmenes: AABB vs. Ball Covering.}
        \end{center}
    \end{columns}
    
    \vspace{0.1cm}
    {\scriptsize \color{PPTGrey} \textit{* Todo obstáculo se aproxima uniendo esferas. Para la PC, esquivar colisiones se simplifica a garantizar que la distancia del robot a cada centro siempre sea mayor al radio.}}
\end{frame}

% Lámina 11: Avance Teórico - Algoritmos y Discretización
{
\usebackgroundtemplate{\includegraphics[width=\paperwidth,height=\paperheight]{figuras/pptx_images/Information_1.jpg}}
\begin{frame}[t]
    \vspace{1.2cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Algoritmos y discretización}} \par
    \vspace{0.4cm}
    \begin{itemize}
        \item \textbf{Voxelización topológica:} Partición de la variedad continua $T^n$ en celdas discretas (Voxeles) para permitir la búsqueda numérica.
        \item \textbf{Criterio métrico:} Análisis comparativo entre Manhattan ($L_1$) y Euclidiana ($L_2$) para guiar la búsqueda en la retícula toroidal \parencite{lavalle2006planning}.
        \item \textbf{Estudio de algoritmos:} Evaluación comparativa entre métodos deterministas (A*) y estocásticos (RRT) sobre la variedad toroidal \parencite{choset2005principles}.
        \item \textbf{Singularidades:} El \enquote{Gimbal-lock} se aborda como una inestabilidad topológica inevitable (Teorema de Farber) \parencite{Mosquera2017}.
    \end{itemize}

    \vspace{0.4cm}
    \begin{block}{Comparativa de métricas de distancia}
        \centering $L_1 = \sum_{i=1}^n |p_i - q_i| \quad \textbf{vs} \quad L_2 = \sqrt{\sum_{i=1}^n (p_i - q_i)^2}$
    \end{block}
\end{frame}
}

    \textbf{\color{PPTSteel} La matemática detrás ($f = g + h$)}
    \begin{itemize}
        \item \textbf{El Pasado ($g$):} Memoria de los pasos exactos que costó llegar al punto actual.
        \item \textbf{La Intuición ($h$):} El cálculo en línea recta de cuánta distancia falta a la meta.
        \item \textbf{El Veredicto ($f$):} La suma final. A* siempre avanza hacia la casilla más "barata".
    \end{itemize}

    \begin{block}{Conclusión}
        Es \textbf{garantizadamente perfecto}, pero puede ser muy lento y consumir mucha memoria si el laberinto está dividido en millones de casillas.
    \end{block}
\end{frame}
}

\usebackgroundtemplate{\includegraphics[width=\paperwidth,height=\paperheight]{figuras/pptx_images/Information_1.jpg}}
\begin{frame}[t]
    \vspace{1.2cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Anexo: Funcionamiento intuitivo del RRT}} \par
    \vspace{0.3cm}
    
    \textbf{\color{PPTSteel} Analogía: Raíces explorando el subsuelo}
    \begin{itemize}
        \item El RRT es libre, no usa cuadrículas. Crece de forma orgánica tirando ramas al azar, priorizando explorar nuevos territorios a gran velocidad.
    \end{itemize}

    \vspace{0.2cm}
    \textbf{\color{PPTSteel} El trío de búsqueda}
    \begin{enumerate}
        \item \textbf{El dardo:} Lanza un punto sorpresa ($q_{rand}$) completamente al azar en el mapa.
        \item \textbf{El imán:} Localiza qué trozo del árbol existente quedó más cerca de ese dardo.
        \item \textbf{El tallo:} Hace crecer una nueva distancia hacia el dardo, de la misma forma que una raíz crece hacia donde intuye humedad.
    \end{enumerate}

    \begin{block}{Conclusión}
        Es \textbf{rápido} para resolver laberintos tridimensionales difíciles, aunque la ruta final puede ser más accidentada y no la más corta.
    \end{block}
\end{frame}

\begin{frame}[t]
    \vspace{1.2cm}
    \hspace{1cm} {\Large \color{PPTSteel} \textbf{Anexo: Métricas Manhattan vs Euclidiana}} \par
    \vspace{0.4cm}
    
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{\color{PPTSteel} L1 (Manhattan)}
        \begin{itemize}
            \item \textbf{Cálculo:} Suma de diferencias absolutas. $|x_1 - x_2| + |y_1 - y_2|$.
            \item \textbf{Ventaja:} Operación ultra-rápida (solo sumas).
            \item \textbf{Visual:} Moverse en forma de escalera o "tablero de ajedrez".
        \end{itemize}
        
        \column{0.5\textwidth}
        \textbf{\color{PPTSteel} L2 (Euclidiana)}
        \begin{itemize}
            \item \textbf{Cálculo:} $\sqrt{(x_1 - x_2)^2 + (y_1 - y_2)^2}$.
            \item \textbf{Ventaja:} Distancia geodésica real (línea recta).
            \item \textbf{Desventaja:} Alto costo de cómputo por raíz cuadrada y potencias.
        \end{itemize}
    \end{columns}
    \vspace{0.2cm}
    \centering \textit{\scriptsize * En el Toro, ambas incluyen la lógica \textbf{wrap-around} para cruzar fronteras.}
\end{frame}

    \begin{columns}[T]
        \column{0.45\textwidth}
        \textbf{\color{PPTSteel} Cinemática Directa (FK)} \\
        \small ¿Dónde está la pinza dados los ángulos?
        \vspace{0.05cm}
        {\scriptsize
        \begin{itemize}
            \item Las variables articulares ($\theta_1, \theta_2, \theta_3$) y las longitudes ($L_1, L_2, L_3$) determinan la pose final trigonométricamente:
            \item $x = L_1 \cos(\theta_1) + L_2 \cos(\theta_1{+}\theta_2) + L_3 \cos(\theta_1{+}\theta_2{+}\theta_3)$
            \item $y = L_1 \sin(\theta_1) + L_2 \sin(\theta_1{+}\theta_2) + L_3 \sin(\theta_1{+}\theta_2{+}\theta_3)$
            \item $\phi = \theta_1 + \theta_2 + \theta_3$
        \end{itemize}
        }
        
        \vspace{0.15cm}
        \textbf{\color{PPTSteel} Cinemática Inversa (IK)} \\
        \small Dada la meta $(x, y, \phi)$, hallar los $\theta_i$.
        \vspace{0.05cm}
        {\scriptsize
        \begin{itemize}
            \item Es un problema algebraico no lineal complejo.
            \item \textbf{Analítico:} Soluciones mediante la Ley de Cosenos (genera ambigüedad codo-arriba/codo-abajo).
            \item \textbf{En nuestro proyecto:} Solo aplicamos IK para el cálculo de la meta; la navegación en sí se deslinda de esta complejidad, resolviéndose puramente en el espacio $\mathcal{C}$ angular.
        \end{itemize}
        }
        
        \column{0.55\textwidth}

            \textit{\scriptsize \textbf{Fig.} Modelo Matemático de un brazo 3R plano.}
